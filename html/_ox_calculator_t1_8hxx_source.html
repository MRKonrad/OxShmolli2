261  template< typename MeasureType >
262  int
263  CalculatorT1<MeasureType>
264  ::prepareToCalculate(){
265 
266  // if fitter does not have to iterate, do not calculate
267  if (this->getFitter()->getMaxFunctionEvals() == 0){
268  return 1; // EXIT_FAILURE
269  }
270 
271  // verify invTimes are sorted
272  for (int i = 0; i < getNSamples()-1; i++){
273  if (getInvTimes()[i] > getInvTimes()[i+1]){
274  throw std::runtime_error("InvTimes have to be sorted!");
275  }
276  }
277 
278  // calculate sign
279  getSignCalculator()->setNSamples(getNSamples());
280  getSignCalculator()->setInvTimes(getInvTimes());
281  getSignCalculator()->setSigMag(getSigMag());
282  getSignCalculator()->setSigPha(getSigPha());
283  getSignCalculator()->setSignal(_Signal);
284  getSignCalculator()->setSigns(_Signs);
285 
286  getSignCalculator()->calculateSign();
287 
288  // calculate start point
289  getStartPointCalculator()->setNSamples(getNSamples());
290  getStartPointCalculator()->setInvTimes(getInvTimes());
291  getStartPointCalculator()->setSigMag(getSigMag());
292  getStartPointCalculator()->setSigns(getSigns());
293  getStartPointCalculator()->setCalculatedStartPoint(_StartPoint);
294 
295  getStartPointCalculator()->calculateStartPoint();
296 
297  return 0; // EXIT_SUCCESS
298  };
