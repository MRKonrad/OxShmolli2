177  int calculateInvCovarianceMatrix(const MeasureType* invTimes, const MeasureType* residuals, const MeasureType* parameters, MeasureType *invCovarianceMatrix) {
178 
179  // invCovarianceMatrix - indexing by column,row
180 
181  int nSamples = this->getNSamples();
182  MeasureType A = parameters[0];
183  MeasureType B = parameters[1];
184  MeasureType T1star = parameters[2];
185  MeasureType T1 = (B/A-1)*T1star;
186 
187  MeasureType dydA = 0;
188  MeasureType dydB = 0;
189  MeasureType dydT1 = 0;
190 
191  for (int i = 0; i < 3*3; ++i) {
192  invCovarianceMatrix[i] = 0;
193  }
194 
195  for (int i = 0; i < nSamples; ++i){
196  MeasureType invTime = invTimes[i];
197  MeasureType myexp = exp ( -invTime * ( B/A - 1) / T1);
198  dydA = 1 - B * myexp * invTime * B / ( T1 * A * A);
199  dydB = -myexp + B * myexp * invTime / ( T1 * A );
200  dydT1 = -B * myexp * invTime * (B/A-1) / ( T1 * T1);
201 
202  invCovarianceMatrix[0] += dydT1 * dydT1; // (0,0)
203  invCovarianceMatrix[1] += dydA * dydT1; // (0,1)
204  invCovarianceMatrix[2] += dydB * dydT1; // (0,2)
205  invCovarianceMatrix[3] += dydT1 * dydA; // (1,0)
206  invCovarianceMatrix[4] += dydA * dydA; // (1,1)
207  invCovarianceMatrix[5] += dydB * dydA; // (1,2)
208  invCovarianceMatrix[6] += dydT1 * dydB; // (2,0)
209  invCovarianceMatrix[7] += dydA * dydB; // (2,1)
210  invCovarianceMatrix[8] += dydB * dydB; // (2,2)
211  }
212 
213  MeasureType SD = KWUtil::calcStandardDeviationArray<MeasureType>(nSamples, residuals);
214 
215  for (int i = 0; i < 3*3; ++i) {
216  invCovarianceMatrix[i] = invCovarianceMatrix[i] / (SD * SD);
217  }
218 
219  return 0; // EXIT_SUCCESS
220  }
