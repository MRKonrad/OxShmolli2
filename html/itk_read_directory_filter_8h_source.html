208  for (unsigned int i = 0; i < reader->GetMetaDataDictionaryArray()->size(); ++i) {
209 
210  const DictionaryType & dictionary = *reader->GetMetaDataDictionaryArray()->at(i);
211 
212  std::string entryId = "0018|0082";
213  DictionaryType::ConstIterator tagItr = dictionary.Find(entryId);
214  MetaDataStringType::ConstPointer entryvalueStr =
215  dynamic_cast<const MetaDataStringType *>(tagItr->second.GetPointer());
216 
217  if (entryvalueStr){

219  invTimesStdVector.push_back(KWUtil::StringToNumber<double>(tagvalue));
