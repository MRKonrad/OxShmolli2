170  ::calculateInvCovarianceMatrix(const MeasureType* invTimes, const MeasureType* residuals, const MeasureType* parameters, MeasureType *invCovarianceMatrix) {
171 
172  // invCovarianceMatrix - indexing by column,row
173 
174  int nSamples = this->getNSamples();
175  MeasureType A = parameters[0];
176  MeasureType B = parameters[1];
177  MeasureType T1star = parameters[2];
178  MeasureType T1 = (B/A-1)*T1star;
179 
180  MeasureType dydA = 0;
181  MeasureType dydB = 0;
182  MeasureType dydT1 = 0;
183 
184  for (int i = 0; i < 3*3; ++i) {
185  invCovarianceMatrix[i] = 0;
186  }
187 
188  for (int i = 0; i < nSamples; ++i){
189  MeasureType invTime = invTimes[i];
190  MeasureType myexp = exp ( -invTime * ( B/A - 1) / T1);
191  dydA = 1 - B * myexp * invTime * B / ( T1 * A * A);
192  dydB = -myexp + B * myexp * invTime / ( T1 * A );
193  dydT1 = -B * myexp * invTime * (B/A-1) / ( T1 * T1);
194 
195  invCovarianceMatrix[0] += dydT1 * dydT1; // (0,0)
196  invCovarianceMatrix[1] += dydA * dydT1; // (0,1)
197  invCovarianceMatrix[2] += dydB * dydT1; // (0,2)
198  invCovarianceMatrix[3] += dydT1 * dydA; // (1,0)
199  invCovarianceMatrix[4] += dydA * dydA; // (1,1)
200  invCovarianceMatrix[5] += dydB * dydA; // (1,2)
201  invCovarianceMatrix[6] += dydT1 * dydB; // (2,0)
202  invCovarianceMatrix[7] += dydA * dydB; // (2,1)
203  invCovarianceMatrix[8] += dydB * dydB; // (2,2)
204  }
205

207 
208  for (int i = 0; i < 3*3; ++i) {
209  invCovarianceMatrix[i] = invCovarianceMatrix[i] / (SD * SD);
210  }

213  }
