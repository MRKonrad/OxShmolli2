229  CalculatorT1Molli<MeasureType>
230  ::calculateInvCovarianceMatrix(const MeasureType* invTimes, const MeasureType* residuals, const MeasureType* parameters, MeasureType *invCovarianceMatrix) {
231 
232  // invCovarianceMatrix - indexing by column,row
233 
234  int nSamples = this->getNSamples();
235  MeasureType A = parameters[0];
236  MeasureType B = parameters[1];
237  MeasureType T1star = parameters[2];
238  MeasureType tempInvCovarianceMatrix[3*3];
239 
240  for (int i = 0; i < 3*3; ++i) {
241  invCovarianceMatrix[i] = 0;
242  tempInvCovarianceMatrix[i] = 0;
243  }
244 
245  if (fabs(A) < std::numeric_limits<MeasureType>::min())
246  return 1; // EXIT_FAILURE
247  if (fabs(T1star) < std::numeric_limits<MeasureType>::min())
248  return 1; // EXIT_FAILURE
249 
250  MeasureType T1 = (B/A-1)*T1star;
251 
252  if (fabs(T1) < std::numeric_limits<MeasureType>::min())
253  return 1; // EXIT_FAILURE
254 
255  MeasureType dydA = 0;
256  MeasureType dydB = 0;
257  MeasureType dydT1 = 0;
258 
259  for (int i = 0; i < nSamples; ++i){
260  MeasureType invTime = invTimes[i];
261  MeasureType myexparg = ( -invTime * ( B/A - 1) / T1);
262 
263  if ((myexparg > std::numeric_limits<MeasureType>::max_exponent)
264  || (myexparg < std::numeric_limits<MeasureType>::min_exponent))
265  return 1; //EXIT_FAILURE
266 
267  MeasureType myexp = exp (myexparg);
268  dydA = 1 - B * myexp * invTime * B / ( T1 * A * A);
269  dydB = -myexp + B * myexp * invTime / ( T1 * A );
270  dydT1 = -B * myexp * invTime * (B/A-1) / ( T1 * T1);
271 
272  tempInvCovarianceMatrix[0] += dydT1 * dydT1; // (0,0)
273  tempInvCovarianceMatrix[1] += dydA * dydT1; // (0,1)
274  tempInvCovarianceMatrix[2] += dydB * dydT1; // (0,2)
275  tempInvCovarianceMatrix[3] += dydT1 * dydA; // (1,0)
276  tempInvCovarianceMatrix[4] += dydA * dydA; // (1,1)
277  tempInvCovarianceMatrix[5] += dydB * dydA; // (1,2)
278  tempInvCovarianceMatrix[6] += dydT1 * dydB; // (2,0)
279  tempInvCovarianceMatrix[7] += dydA * dydB; // (2,1)
280  tempInvCovarianceMatrix[8] += dydB * dydB; // (2,2)
281  }
282 
283  MeasureType SD = KWUtil::calcStandardDeviationArray<MeasureType>(nSamples, residuals);
284 
285  if (fabs(SD) < std::numeric_limits<MeasureType>::min())
286  return 1; //EXIT_FAILURE
287 
288  for (int i = 0; i < 3*3; ++i) {
289  tempInvCovarianceMatrix[i] = tempInvCovarianceMatrix[i] / (SD * SD);
290  }
291 
292  for (int i = 0; i < 3*3; ++i) {
293  invCovarianceMatrix[i] = tempInvCovarianceMatrix[i];
294  }
295 
296  return 0; // EXIT_SUCCESS
297  }
