167  std::vector <std::string> fileContent;
168  std::string line;
169

171  std::ifstream myfile(filePath);
172  if (myfile.is_open()) {
173  while (getline(myfile, line)) {
174  fileContent.push_back(line);
175  }
176  myfile.close();
177  } else {

179  }

182  }
183  return fileContent;
